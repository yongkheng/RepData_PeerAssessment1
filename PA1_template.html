fn            <- function(x){total_steps[total_steps$interval == x,]$means}
filled_dt     <- tbl_df(data)
nas           <- filled_dt[is.na(filled_dt),]
p             <- sapply(nas$interval, fn)
filled_dt[is.na(filled_dt)] <- p
